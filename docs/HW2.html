Each spot, with a diameter of 55 microns in Visium, typically encompasses multiple cells. Instead of annotating individual spots directly, deconvolution is used to estimate the cell-type fractions within each spot. Deconvolution can be performed with or without single-cell RNA-seq data as a reference.

import scanpy as sc
import pandas as pd
import numpy as np
import json
from PIL import Image

# 1. load h5ad
adata = sc.read_h5ad("path/to/your/file.h5ad")

# 2. load spatial coordinate
# 2.1 load in tissue_positions.csv
tissue_positions_path = "spatial/GSM8399479_S22_Sample30_C2_Blu_tissue_positions.csv"
positions = pd.read_csv(tissue_positions_path, header=None)
positions.columns = ["barcode", "in_tissue", "array_row", "array_col", "pxl_row_in_fullres", "pxl_col_in_fullres"]
positions.set_index("barcode", inplace=True)

# add coordinate to .obs
adata.obs = adata.obs.merge(positions, left_index=True, right_index=True, how="left")

# 2.2 load scalefactors_json.json
scalefactors_path = "spatial/GSM8399479_S22_Sample30_C2_Blu_scalefactors_json.json"
with open(scalefactors_path, "r") as f:
    scalefactors = json.load(f)

# save scale factor to .uns
adata.uns["spatial"] = {"scalefactors": scalefactors}

# 3. load images
image_path = "spatial/GSM8399479_S22_Sample30_C2_Blu_tissue_hires_image.png"
image = Image.open(image_path)
adata.uns["spatial"]["hires_image"] = np.array(image)

# 4. integrate coordinate to obsm
adata.obsm["spatial"] = adata.obs[["pxl_row_in_fullres", "pxl_col_in_fullres"]].to_numpy()

# 5. check adata
print(adata)
